46impl GprIndex {
47    /// Get register index from raw value.
48    pub fn from_raw(raw: u32) -> Option<Self> {
49        use GprIndex::*;
50        let index = match raw {
51            0 => Zero,
52            1 => RA,
53            2 => SP,
54            3 => GP,
55            4 => TP,
56            5 => T0,
57            6 => T1,
58            7 => T2,
59            8 => S0,
60            9 => S1,
61            10 => A0,
62            11 => A1,
63            12 => A2,
64            13 => A3,
65            14 => A4,
66            15 => A5,
67            16 => A6,
68            17 => A7,
69            18 => S2,
70            19 => S3,
71            20 => S4,
72            21 => S5,
73            22 => S6,
74            23 => S7,
75            24 => S8,
76            25 => S9,
77            26 => S10,
78            27 => S11,
79            28 => T3,
80            29 => T4,
81            30 => T5,
82            31 => T6,
83            _ => {
84                return None;
85            }
86        };
87        Some(index)
88    }

91impl GeneralPurposeRegisters {
92    /// Returns the value of the given register.
93    pub fn reg(&self, reg_index: GprIndex) -> usize {
94        self.0[reg_index as usize]
95    }
96
97    /// Sets the value of the given register.
98    pub fn set_reg(&mut self, reg_index: GprIndex, val: usize) {
99        if reg_index == GprIndex::Zero {
100            return;
101        }
102
103        self.0[reg_index as usize] = val;
104    }
105
106    /// Returns the argument registers.
107    /// This is avoids many calls when an SBI handler needs all of the argmuent regs.
108    pub fn a_regs(&self) -> &[usize] {
109        &self.0[GprIndex::A0 as usize..=GprIndex::A7 as usize]
110    }
111
112    /// Returns the arguments register as a mutable.
113    pub fn a_regs_mut(&mut self) -> &mut [usize] {
114        &mut self.0[GprIndex::A0 as usize..=GprIndex::A7 as usize]
115    }
116}

141/// The CSRs that are only in effect when virtualization is enabled (V=1) and must be saved and
142/// restored whenever we switch between VMs.
143#[derive(Debug, Default, Clone)]
144#[repr(C)]
145pub struct GuestVsCsrs {
146    pub htimedelta: usize,
147    pub vsstatus: usize,
148    pub vsie: usize,
149    pub vstvec: usize,
150    pub vsscratch: usize,
151    pub vsepc: usize,
152    pub vscause: usize,
153    pub vstval: usize,
154    pub vsatp: usize,
155    pub vstimecmp: usize,
156}
157
158impl GuestVsCsrs {
159    // Load the VS-level CSRs from hardware into this structure.
160    pub fn load_from_hw(&mut self) {
161        use riscv_h::register::{
162            htimedelta, vsatp, vscause, vsepc, vsie, vsscratch, vsstatus, vstval, vstvec,
163        };
164
165        self.htimedelta = htimedelta::read();
166        self.vsstatus = vsstatus::read().bits();
167        self.vsie = vsie::read().bits();
168        self.vstvec = vstvec::read().bits();
169        self.vsscratch = vsscratch::read();
170        self.vsepc = vsepc::read();
171        self.vscause = vscause::read().bits();
172        self.vstval = vstval::read();
173        self.vsatp = vsatp::read().bits();
174        // vstimecmp is not a CSR but a memory-mapped register.
175    }
176}

199/// CSRs written on an exit from virtualization that are used by the hypervisor to determine the cause
200/// of the trap.
201#[derive(Debug, Default, Clone)]
202#[repr(C)]
203pub struct VmCpuTrapState {
204    pub scause: usize,
205    pub stval: usize,
206    pub htval: usize,
207    pub htinst: usize,
208}
209
210impl VmCpuTrapState {
211    /// Reads the trap-related CSRs from hardware and stores them in this structure.
212    pub fn load_from_hw(&mut self) {
213        use riscv::register::{scause, stval};
214        use riscv_h::register::{htinst, htval};
215
216        self.scause = scause::read().bits();
217        self.stval = stval::read();
218        self.htval = htval::read();
219        self.htinst = htinst::read();
220    }
221
222    /// Returns the guest physical address that caused a guest page fault.
223    ///
224    /// Make sure [`Self::load_from_hw`] has been called before using this method.
225    pub fn gpt_page_fault_addr(&self) -> GuestPhysAddr {
226        let pfa = (self.htval << 2) | (self.stval & 0b11);
227        pfa.into()
228    }
229}
