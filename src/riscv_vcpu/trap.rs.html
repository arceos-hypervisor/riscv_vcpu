7/// Exception
8#[derive(Copy, Clone, Debug, Eq, PartialEq)]
9#[repr(usize)]
10pub enum Exception {
11    InstructionMisaligned = 0,
12    InstructionFault = 1,
13    IllegalInstruction = 2,
14    Breakpoint = 3,
15    LoadMisaligned = 4,
16    LoadFault = 5,
17    StoreMisaligned = 6,
18    StoreFault = 7,
19    UserEnvCall = 8,
20    SupervisorEnvCall = 9,
21    VirtualSupervisorEnvCall = 10,
22    InstructionPageFault = 12,
23    LoadPageFault = 13,
24    StorePageFault = 15,
25    InstructionGuestPageFault = 20,
26    LoadGuestPageFault = 21,
27    VirtualInstruction = 22,
28    StoreGuestPageFault = 23,
29}

31/// SAFETY: `Exception` represents the standard RISC-V exceptions
32unsafe impl ExceptionNumber for Exception {
33    const MAX_EXCEPTION_NUMBER: usize = Self::StoreGuestPageFault as usize;
34
35    #[inline]
36    fn number(self) -> usize {
37        self as usize
38    }
39
40    #[inline]
41    fn from_number(value: usize) -> Result<Self> {
42        match value {
43            0 => Ok(Self::InstructionMisaligned),
44            1 => Ok(Self::InstructionFault),
45            2 => Ok(Self::IllegalInstruction),
46            3 => Ok(Self::Breakpoint),
47            4 => Ok(Self::LoadMisaligned),
48            5 => Ok(Self::LoadFault),
49            6 => Ok(Self::StoreMisaligned),
50            7 => Ok(Self::StoreFault),
51            8 => Ok(Self::UserEnvCall),
52            9 => Ok(Self::SupervisorEnvCall),
53            10 => Ok(Self::VirtualSupervisorEnvCall),
54            12 => Ok(Self::InstructionPageFault),
55            13 => Ok(Self::LoadPageFault),
56            15 => Ok(Self::StorePageFault),
57            20 => Ok(Self::InstructionGuestPageFault),
58            21 => Ok(Self::LoadGuestPageFault),
59            22 => Ok(Self::VirtualInstruction),
60            23 => Ok(Self::StoreGuestPageFault),
61            _ => Err(Error::InvalidVariant(value)),
62        }
63    }
64}

66#[allow(dead_code)]
67const fn hyp_gpr_offset(index: GprIndex) -> usize {
68    offset_of!(VmCpuRegisters, hyp_regs)
69        + offset_of!(HypervisorCpuState, gprs)
70        + (index as usize) * size_of::<u64>()
71}
72
73#[allow(dead_code)]
74const fn guest_gpr_offset(index: GprIndex) -> usize {
75    offset_of!(VmCpuRegisters, guest_regs)
76        + offset_of!(GuestCpuState, gprs)
77        + (index as usize) * size_of::<u64>()
78}
